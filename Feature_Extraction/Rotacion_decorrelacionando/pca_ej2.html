def mypca(X,n_components=0):
    # computing eigenvalues and eigenvectors of covariance matrix from X data
    # D: eigenvalues, U: eigenvectors, Y: projections
    from numpy import mean,cov,dot,linalg,argsort
    n,p=X.shape
    # subtract the mean
    mu = mean(X,axis=0)
    X_c = X-mu
    #unbiased covariance estimate
    C = cov(X_c, rowvar=False)
    #eigendecomposition
    D, U = linalg.eigh(C)
    # sorting the eigenvalues
    # in descending order
    idx = argsort(-D)
    # sorting eigenvectors according to the sorted eigenvalues
    U = U[:, idx]
    # sorting eigenvalues
    D = D[idx]
    if n_components < p and n_components > 0:
        D = D[:n_components]
        U = U[:, :n_components] # cutting some PCs if needed
    # projection of the data in the new space
    Y = dot(U.T, X_c.T).T
    return D,U,Y,mu

# sklearn PCA
pca = PCA(n_components = 3, svd_solver='full')
pca.fit(X)
D = pca.explained_variance_
U = pca.components_.T
mu = pca.mean_
perc = pca.explained_variance_ratio_
n_components = pca.n_components_
Y = pca.transform(X)
print('SKLEARN PCA')
print('\nD=\n',D,'\nU=\n',U,'\nmu=\n',mu,'\nperc=\n',perc,'\nY=\n',Y.shape)

# Mi PCA
D,U,Y,mu=mypca(X,n_components=0)

print('\nD=\n',D,'\nU=\n',U,'\nmu=\n',mu,'\nperc=\n',perc,'\nY=\n',Y.shape)
