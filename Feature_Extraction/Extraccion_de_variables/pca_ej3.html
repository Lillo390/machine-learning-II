#PCA
pca = PCA(n_components = 3, svd_solver='full')
pca.fit(X)
D = pca.explained_variance_
U = pca.components_.T
mu = pca.mean_
perc = pca.explained_variance_ratio_
n_components = pca.n_components_
Y = pca.transform(X)
print(U.shape)
